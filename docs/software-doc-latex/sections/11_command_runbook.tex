\section{Appendix: Command-Only Runbook}

\subsection{Local Studio: Service and Basic Checks}
\begin{lstlisting}[language=bash]
cd ~/Desktop/github/agri-app
./deploy_rpi.sh --reload-local-server
curl -s http://127.0.0.1:5000/api/v1/health
\end{lstlisting}

\subsection{RPi: Sync and Backend Reload}
\begin{lstlisting}[language=bash]
cd ~/Desktop/github/agri-app
./deploy_rpi.sh --sync
./deploy_rpi.sh --reload-server
\end{lstlisting}

\subsection{RPi: Service Management}
\begin{lstlisting}[language=bash]
systemctl --user status agriapp-server.service
systemctl --user restart agriapp-server.service
journalctl --user -u agriapp-server.service -f
\end{lstlisting}

\subsection{RPi: One-shot Capture API}
\begin{lstlisting}[language=bash]
curl -X POST http://127.0.0.1:5000/api/v1/capture/rpi/oneshot
\end{lstlisting}

\subsection{RPi: Loop Capture API}
\begin{lstlisting}[language=bash]
curl http://127.0.0.1:5000/api/v1/capture/loop/status
curl -X POST http://127.0.0.1:5000/api/v1/capture/loop/start \
  -H "Content-Type: application/json" \
  -d '{"interval_seconds":300}'
curl -X POST http://127.0.0.1:5000/api/v1/capture/loop/stop
\end{lstlisting}

\subsection{RPi: Sensor Diagnostics}
\begin{lstlisting}[language=bash]
# Stop server first to avoid serial contention
systemctl --user stop agriapp-server.service

cd ~/agri-app
python3 test/test-gps.py --port /dev/serial0 --baud 9600 --watch --raw
python3 test/test-bme.py

# Restart server
systemctl --user start agriapp-server.service
\end{lstlisting}
